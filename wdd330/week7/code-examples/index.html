JavaScript
function imaginarySquareRoot(number) {
    'use strict';
    let answer;
    try {
        answer = String(squareRoot(number));
    }
    catch(error) {
        answer = squareRoot(-number)+"i";
    }
    finally {
        return `+ or - ${answer}`;
    }
}

I don't know if I'd ever use this, but it is nice to know that it is an available tool.
